  function validateForm() {
    const name = document.getElementById("patientName").value.trim();
    const email = document.getElementById("patientEmail").value.trim();
    const phone = document.getElementById("patientPhone").value.trim();
    const doctor = document.getElementById("doctor").value;

    if (!name || !email || !phone || !doctor) {
      alert("Please fill in all fields.");
      return false;
    }

    if (!/^\S+@\S+\.\S+$/.test(email)) {
      alert("Enter a valid email.");
      return false;
    }

    if (!/^\+?\d{9,15}$/.test(phone)) {
      alert("Enter a valid phone number.");
      return false;
    }

    alert("Appointment request sent successfully!");
    return true;
  }

  const toggleButton = document.getElementById('toggleAmbulanceForm');
  const ambulanceFormSection = document.getElementById('ambulanceFormSection');

  toggleButton.addEventListener('click', () => {
    if (ambulanceFormSection.style.display === 'block') {
      ambulanceFormSection.style.display = 'none';
      toggleButton.textContent = 'Ambulance Support';
    } else {
      ambulanceFormSection.style.display = 'block';
      toggleButton.textContent = 'Hide Ambulance Form';
      ambulanceFormSection.scrollIntoView({behavior: 'smooth'});
    }
  });

  document.getElementById('ambulanceForm').addEventListener('submit', function(e) {
    e.preventDefault();

    const name = document.getElementById('ambName').value.trim();
